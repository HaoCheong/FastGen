from unit.conftest import client, SUCCESS
import json

def unpack(function):
    ''' Wrapper to unpack the json values into parsable dictionary. Easier for testing '''
    def get_data(*args):
        resp = function(*args)
        if resp.status_code != SUCCESS:
            data = json.loads(resp.text)
            return {'status': resp.status_code,
                    'detail': data['detail']
                    }
        else:
            return {'status': resp.status_code,
                    'data': json.loads(resp.text)
                    }
    return get_data
